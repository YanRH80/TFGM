% ── Typography ────────────────────────────────────────────────

% ── Spacing ───────────────────────────────────────────────────
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.7em}

% ── Links ─────────────────────────────────────────────────────
\usepackage{hyperref}

% ── Section titles (use palette colors) ───────────────────────
\usepackage{titlesec}
\titleformat{\section}
  {\color{PrimaryDark}\sffamily\Large\bfseries}
  {\thesection}{1em}{}[\vspace{0em}{\color{AccentLight}\titlerule}]
\titlespacing*{\section}{0pt}{2.5em}{1em}

\titleformat{\subsection}
  {\color{PrimaryDark}\sffamily\large\bfseries}
  {\thesubsection}{1em}{}

\usepackage{caption}
\captionsetup{

  labelsep=space, format=hang
}

\setlist[itemize]{label=\textcolor{AccentLight}{\rule{0.8ex}{0.8ex}}, leftmargin=*, topsep=0pt}

\tcbuselibrary{skins, breakable}

  enhanced, breakable,
  colback=white,
  colframe=AccentLight,
  colbacktitle=AccentLight,
  coltitle=black,
  fonttitle=\bfseries\rmfamily\large,

  boxed title style={frame hidden, arc=0pt, outer arc=0pt,
    top=4pt, bottom=4pt, left=8pt, right=8pt}
}
